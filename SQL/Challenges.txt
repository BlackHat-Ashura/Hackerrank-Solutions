SELECT H.HACKER_ID, H.NAME, COUNT(C.CHALLENGE_ID) AS C_COUNTER
FROM HACKERS H, CHALLENGES C
WHERE
H.HACKER_ID = C.HACKER_ID
GROUP BY H.HACKER_ID, H.NAME
HAVING
C_COUNTER IN (SELECT TEMP.COUNTER
             FROM (SELECT HACKER_ID, COUNT(CHALLENGE_ID) AS COUNTER
                  FROM CHALLENGES
                  GROUP BY HACKER_ID) AS TEMP
              GROUP BY TEMP.COUNTER
              HAVING COUNT(TEMP.COUNTER)=1
             ) OR
C_COUNTER = (SELECT MAX(TEMP.COUNTER)
             FROM (SELECT HACKER_ID, COUNT(CHALLENGE_ID) AS COUNTER
                  FROM CHALLENGES
                  GROUP BY HACKER_ID) AS TEMP
             )
ORDER BY C_COUNTER DESC, H.HACKER_ID ASC;
